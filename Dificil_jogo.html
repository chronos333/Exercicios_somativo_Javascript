// --- Seleção de Elementos do DOM --- const botoes = document.querySelectorAll('button'); // Captura todos os botões de escolha const displayResultado = document.querySelector('h4'); // Onde exibiremos o texto final // Lista de opções possíveis para facilitar a lógica do computador const opcoes = ['pedra', 'papel', 'tesoura']; // --- Gerenciamento de Eventos --- // Adiciona um evento de clique para cada botão individualmente botoes.forEach(botao => { botao.addEventListener('click', () => { const escolhaUsuario = botao.id; // Pega o ID (pedra, papel ou tesoura) const escolhaComputador = gerarEscolhaComputador(); // Sorteia a jogada do PC // Busca os emojis correspondentes às escolhas const emojiUser = retornarEmoji(escolhaUsuario); const emojiPC = retornarEmoji(escolhaComputador); // Calcula quem venceu a rodada const statusWin = determinarVencedor(escolhaUsuario, escolhaComputador); // Atualiza a interface com as informações processadas exibirResultado(escolhaUsuario, escolhaComputador, statusWin, emojiUser, emojiPC); }); }); // --- Funções de Lógica --- /** * Converte a string de escolha em um emoji visual. * @param {string} escolha - A jogada feita. */ function retornarEmoji(escolha) { if (escolha === 'pedra') return "🪨"; if (escolha === 'papel') return "📄"; if (escolha === 'tesoura') return "✂️"; } /** * Gera uma escolha aleatória para o computador baseada no array 'opcoes'. */ function gerarEscolhaComputador() { const indiceAleatorio = Math.floor(Math.random() * 3); // Gera 0, 1 ou 2 return opcoes[indiceAleatorio]; } /** * Compara as jogadas e define o estado da partida. */ function determinarVencedor(usuario, computador) { // Caso de empate: escolhas idênticas if (usuario === computador) return "Empate! 🤝"; // Regras de vitória do usuário if ( (usuario === 'pedra' && computador === 'tesoura') || // Pedra vence Tesoura (usuario === 'papel' && computador === 'pedra') || // Papel vence Pedra (usuario === 'tesoura' && computador === 'papel') // Tesoura vence Papel ) { return "Você Venceu! 🎉"; } else { // Se não empatou nem ganhou, o usuário perdeu return "Você Perdeu! 😢"; } } /** * Atualiza o conteúdo HTML da página para mostrar o resultado final. */ function exibirResultado(user, pc, res, emojuser, emojpc) { displayResultado.innerHTML = ` Sua escolha: ${user.toUpperCase()} ${emojuser} | Computador: ${pc.toUpperCase()} ${emojpc} |
${res} `; }
